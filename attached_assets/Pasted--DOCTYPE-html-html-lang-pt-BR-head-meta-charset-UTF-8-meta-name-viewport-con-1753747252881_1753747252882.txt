// Funções de navegação
function mostrarView(view) {
    document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));
    document.getElementById('view-' + view).classList.add('active');
    
    // Carregar dados específicos para cada view
    switch(view) {
        case 'consultar':
            carregarTodosRelatorios();
            break;
        case 'listar':
            carregarListaCompleta();
            break;
        case 'imprimir':
            carregarListaImpressao();
            break;
    }
}

// Inicializar formulário com linhas vazias
function inicializarFormulario() {
    // Adicionar 5 linhas para equipamentos
    const equipamentosBody = document.getElementById('equipamentosTableBody');
    if(equipamentosBody) {
        equipamentosBody.innerHTML = '';
        for(let i = 0; i < 5; i++) {
            adicionarLinhaEquipamento();
        }
    }

    // Adicionar 4 linhas para peças
    const pecasBody = document.getElementById('pecasTableBody');
    if(pecasBody) {
        pecasBody.innerHTML = '';
        for(let i = 0; i < 4; i++) {
            adicionarLinhaPeca();
        }
    }

    // Definir data atual
    const dataInput = document.querySelector('input[name="dataServico"]');
    if(dataInput) {
        dataInput.value = new Date().toISOString().split('T')[0];
    }
}

    const tbody = document.getElementById('equipamentosTableBody');
    if(!tbody) return;

function calcularTotalPecas() {
    const quantidades = document.querySelectorAll('input[name="quantidade[]"]');
    const valoresUnitarios = document.querySelectorAll('input[name="valorUnitario[]"]');
    const valoresTotal = document.querySelectorAll('input[name="valorTotal[]"]');
    const totalGeral = document.querySelector('input[name="totalPecas"]');
    
    let somaTotal = 0;
    
    for(let i = 0; i < quantidades.length; i++) {
        const qtd = parseFloat(quantidades[i].value) || 0;
        const vlrUnit = parseFloat(valoresUnitarios[i].value) || 0;
        const total = qtd * vlrUnit;
        
        valoresTotal[i].value = total.toFixed(2);
        somaTotal += total;
    }
    
    if(totalGeral) {
        totalGeral.value = somaTotal.toFixed(2);
    }
}

// Função principal para salvar relatório
function salvarRelatorio() {
    const form = document.getElementById('formCriarRelatorio');
    if(!form) return;
    
    const formData = new FormData(form);
    const relatorio = {
        id: proximoId++,
        numero: String(proximoId - 1).padStart(3, '0'),
        tiposServico: [],
        equipamentos: [],
        pecas: [],
        dataCriacao: new Date().toISOString(),
    };

    // Capturar tipos de serviço
    document.querySelectorAll('input[name="tipoServico"]:checked').forEach(checkbox => {
        relatorio.tiposServico.push(checkbox.value);
    });

    // Capturar dados básicos
    relatorio.razaoSocial = formData.get('razaoSocial') || '';
    relatorio.cnpj = formData.get('cnpj') || '';
    relatorio.endereco = formData.get('endereco') || '';
    relatorio.cidadeUf = formData.get('cidadeUf') || '';
    relatorio.inscricaoEstadual = formData.get('inscricaoEstadual') || '';
    relatorio.servicosExecutados = formData.get('servicosExecutados') || '';
    relatorio.observacoes = formData.get('observacoes') || '';
    relatorio.nomeTecnico = formData.get('nomeTecnico') || '';
    relatorio.rgTecnico = formData.get('rgTecnico') || '';
    relatorio.cpfTecnico = formData.get('cpfTecnico') || '';
    relatorio.horarioChegada = formData.get('horarioChegada') || '';
    relatorio.horarioSaida = formData.get('horarioSaida') || '';
    relatorio.dataServico = formData.get('dataServico') || '';
    relatorio.assinaturaCarimbo = formData.get('assinaturaCarimbo') || '';

    // Capturar testes
    relatorio.testes = {
        etc: formData.get('etc') || '',
        eta: formData.get('eta') || '',
        gc: formData.get('gc') || '',
        ga: formData.get('ga') || '',
        arla: formData.get('arla') || '',
        dc: formData.get('dc') || '',
        da: formData.get('da') || ''
    };

    // Capturar equipamentos
    const nBicos = formData.getAll('nBico[]');
    const marcas = formData.getAll('marca[]');
    const modelos = formData.getAll('modelo[]');
    const series = formData.getAll('serie[]');
    const produtos = formData.getAll('produto[]');
    const inmetros = formData.getAll('inmetro[]');
    const portarias = formData.getAll('portaria[]');
    const lacresRetirados = formData.getAll('lacreRetirado[]');
    const lacresColocados = formData.getAll('lacreColocado[]');
    const selosRetirados = formData.getAll('seloRetirado[]');
    const selosColocados = formData.getAll('seloColocado[]');

    for(let i = 0; i < nBicos.length; i++) {
        if(nBicos[i] || marcas[i] || modelos[i]) {
            relatorio.equipamentos.push({
                nBico: nBicos[i] || '',
                marca: marcas[i] || '',
                modelo: modelos[i] || '',
                serie: series[i] || '',
                produto: produtos[i] || '',
                inmetro: inmetros[i] || '',
                portaria: portarias[i] || '',
                lacreRetirado: lacresRetirados[i] || '',
                lacreColocado: lacresColocados[i] || '',
                seloRetirado: selosRetirados[i] || '',
                seloColocado: selosColocados[i] || ''
            });
        }
    }

    // Capturar peças
    const pecas = formData.getAll('peca[]');
    const quantidades = formData.getAll('quantidade[]');
    const valoresUnitarios = formData.getAll('valorUnitario[]');
    const valoresTotal = formData.getAll('valorTotal[]');

    for(let i = 0; i < pecas.length; i++) {
        if(pecas[i] || quantidades[i]) {
            relatorio.pecas.push({
                peca: pecas[i] || '',
                quantidade: quantidades[i] || '',
                valorUnitario: valoresUnitarios[i] || '',
                valorTotal: valoresTotal[i] || ''
            });
        }
    }

    relatorio.totalPecas = formData.get('totalPecas') || '0.00';

    // Salvar no array
    relatorios.push(relatorio);

    // Mostrar sucesso
    mostrarModalSucesso(`Relatório N° ${relatorio.numero} salvo com sucesso!`);

    // Limpar formulário e gerar novo número
    limparFormulario();
    gerarNumeroRelatorio();
}

function limparFormulario() {
    const form = document.getElementById('formCriarRelatorio');
    if(form) {
        form.reset();
        const dataServico = document.querySelector('input[name="dataServico"]');
        if(dataServico) {
            dataServico.value = new Date().toISOString().split('T')[0];
        }
        
        const totalPecas = document.querySelector('input[name="totalPecas"]');
        if(totalPecas) {
            totalPecas.value = '';
        }
        
        // Limpar valores calculados
        document.querySelectorAll('input[name="valorTotal[]"]').forEach(input => {
            input.value = '';
        });
    }
}

function atualizarRelatorio() {

    const formData = new FormData(form);
    const relatorioId = parseInt(formData.get('relatorioId'));
    
    const relatorioIndex = relatorios.findIndex(rel => rel.id === relatorioId);
    if(relatorioIndex === -1) {
        mostrarModalErro('Relatório não encontrado para atualização.');
        return;
    }

    // Atualizar dados
    const relatorio = relatorios[relatorioIndex];
    
    // Capturar tipos de serviço
    relatorio.tiposServico = [];
    document.querySelectorAll('#formEdicaoRelatorio input[name="tipoServico"]:checked').forEach(checkbox => {
        relatorio.tiposServico.push(checkbox.value);
    });

    // Atualizar campos básicos
    relatorio.razaoSocial = formData.get('razaoSocial') || '';
    relatorio.cnpj = formData.get('cnpj') || '';
    relatorio.endereco = formData.get('endereco') || '';
    relatorio.cidadeUf = formData.get('cidadeUf') || '';
    relatorio.inscricaoEstadual = formData.get('inscricaoEstadual') || '';
    relatorio.servicosExecutados = formData.get('servicosExecutados') || '';
    relatorio.observacoes = formData.get('observacoes') || '';
    relatorio.nomeTecnico = formData.get('nomeTecnico') || '';
    relatorio.rgTecnico = formData.get('rgTecnico') || '';
    relatorio.dataServico = formData.get('dataServico') || '';

// Funções de exclusão
function buscarRelatorioParaExclusao() {
    const numeroElement = document.getElementById('numeroOrdemExcluir');
    if(!numeroElement) return;
    
    const numero = numeroElement.value.trim();
    
    if(!numero) {
        mostrarModalErro('Por favor, digite o número da ordem de serviço.');
        return;
    }

    const relatorio = relatorios.find(rel => rel.numero === numero.padStart(3, '0'));
    
    if(!relatorio) {
        mostrarModalErro(`Relatório N° ${numero} não encontrado.`);
        return;
    }

    // Mostrar dados do relatório para confirmação
    mostrarDadosParaExclusao(relatorio);
    const container = document.getElementById('relatorioExclusaoContainer');
    if(container) {
        container.style.display = 'block';
    }
}

function mostrarDadosParaExclusao(relatorio) {
    const container = document.getElementById('relatorioExclusaoContainer');
    if(!container) return;
    
    container.innerHTML = `
        <div class="alert alert-warning">
            <i class="bi bi-exclamation-triangle me-2"></i>
            <strong>ATENÇÃO:</strong> Você está prestes a excluir o relatório abaixo.
        </div>
        
        <div class="card">
            <div class="card-header bg-danger text-white">
                <h5 class="mb-0">Relatório N° ${relatorio.numero}</h5>
            </div>
            <div class="card-body">
                <div class="row">
                    <div class="col-md-6">
                        <p><strong>Razão Social:</strong> ${relatorio.razaoSocial}</p>
                        <p><strong>CNPJ:</strong> ${relatorio.cnpj}</p>
                        <p><strong>Técnico:</strong> ${relatorio.nomeTecnico}</p>
                    </div>
                    <div class="col-md-6">
                        <p><strong>Data Serviço:</strong> ${formatarData(relatorio.dataServico)}</p>
                        <p><strong>Data Criação:</strong> ${formatarDataHora(relatorio.dataCriacao)}</p>
                        <p><strong>Tipos:</strong> ${relatorio.tiposServico.join(', ').toUpperCase()}</p>
                    </div>
                </div>
            </div>
        </div>
        
        <input type="hidden" id="relatorioParaExcluir" value="${relatorio.id}">
        
        <div class="d-grid gap-2 d-md-flex justify-content-md-end mt-4">
            <button type="button" class="btn btn-outline-secondary me-md-2" onclick="cancelarExclusao()">
                <i class="bi bi-x-circle me-1"></i>Cancelar
            </button>
            <button type="button" class="btn btn-danger" onclick="confirmarExclusao()">
                <i class="bi bi-trash me-1"></i>Excluir Definitivamente
            </button>
        </div>
    `;
}

function confirmarExclusao() {
    const modal = document.getElementById('modalConfirmacaoExclusao');
    if(modal && window.bootstrap) {
        new bootstrap.Modal(modal).show();
    }
}

function executarExclusao() {
    const relatorioIdElement = document.getElementById('relatorioParaExcluir');
    if(!relatorioIdElement) return;
    
    const relatorioId = parseInt(relatorioIdElement.value);
    const relatorioIndex = relatorios.findIndex(rel => rel.id === relatorioId);
    
    if(relatorioIndex === -1) {
        mostrarModalErro('Relatório não encontrado para exclusão.');
        return;
    }

    const relatorio = relatorios[relatorioIndex];
    relatorios.splice(relatorioIndex, 1);
    
    mostrarModalSucesso(`Relatório N° ${relatorio.numero} excluído com sucesso!`);
    cancelarExclusao();
}

function cancelarExclusao() {
    const container = document.getElementById('relatorioExclusaoContainer');
    if(container) {
        container.style.display = 'none';
    }
    
    const numeroInput = document.getElementById('numeroOrdemExcluir');
    if(numeroInput) {
        numeroInput.value = '';
    }
}

// Funções utilitárias
function formatarData(dataISO) {
    if(!dataISO) return '';
    const data = new Date(dataISO);
    return data.toLocaleDateString('pt-BR');
}

function formatarDataHora(dataISO) {
    if(!dataISO) return '';
    const data = new Date(dataISO);
    return data.toLocaleDateString('pt-BR') + ' às ' + data.toLocaleTimeString('pt-BR');
}

function mostrarModalSucesso(mensagem) {
    const modalBody = document.getElementById('modalSucessoBody');
    if(modalBody) {
        modalBody.textContent = mensagem;
    }
    
    const modal = document.getElementById('modalSucesso');
    if(modal && window.bootstrap) {
        new bootstrap.Modal(modal).show();
    }
}

function mostrarModalErro(mensagem) {
    const modalBody = document.getElementById('modalErroBody');
    if(modalBody) {
        modalBody.textContent = mensagem;
    }
    
    const modal = document.getElementById('modalErro');
    if(modal && window.bootstrap) {
        new bootstrap.Modal(modal).show();
    }
}
